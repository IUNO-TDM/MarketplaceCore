Code

CREATE FUNCTION getallattributes(vuseruuid uuid, vroles text[]) RETURNS TABLE(attributeuuid uuid, attributename character varying, createdat timestamp with time zone, createdby uuid, updatedat timestamp with time zone, updatedby uuid)
    LANGUAGE plpgsql
    AS $$

    DECLARE
        vFunctionName varchar := 'GetAllAttributes';
        vIsAllowed boolean := (select public.checkPermissions(vRoles, vFunctionName));

    BEGIN

    IF(vIsAllowed) THEN

    RETURN QUERY    (SELECT  attributeuuid,
                attributename,
                att.createdat at time zone 'utc',
                att.createdby,
                att.updatedat at time zone 'utc',
                att.updatedby
            FROM attributes att
        );

    ELSE
         RAISE EXCEPTION '%', 'Insufficiency rigths';
         RETURN;
    END IF;

    END;
    $$;
